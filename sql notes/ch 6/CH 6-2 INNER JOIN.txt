1 DISPLAY FIRST_NAME , DEPARTMENT_NAME

	SELECT FIRST_NAME , DEPARTMENT_NAME FROM EMPLOYEES INNER JOIN DEPARTMENTS 							USING(DEPARTMENT_ID);
	SELECT FIRST_NAME , DEPARTMENT_NAME FROM EMPLOYEES E INNER JOIN DEPARTMENTS D 					ON(D.DEPARTMENT_ID=E.DEPARTMENT_ID);

	INNER JOIN REGIONS R USING(REGION_ID);
