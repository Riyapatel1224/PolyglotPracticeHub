1 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , HIGHEST SALARY PAID TO THE CORRESPONDING DEPARTMENT_ID

SELECT FIRST_NAME, DEPARTMENT_ID, SALARY, "MAX(SALARY)"
FROM EMPLOYEES JOIN (SELECT MAX(SALARY),DEPARTMENT_ID FROM EMPLOYEES GROUP BY DEPARTMENT_ID) USING(DEPARTMENT_ID);

2 DISPLAY LAST_NAME , JOB_ID , SALARY , LOWEST SALARY PAID TO THE CORRESPONDING JOB_ID

SELECT LAST_NAME, JOB_ID, SALARY, "MIN(SALARY)"
FROM EMPLOYEES JOIN (SELECT JOB_ID, MIN(SALARY) FROM EMPLOYEES GROUP BY JOB_ID) USING(JOB_ID);

3 DISPLAY FIRST_NAME , MANAGER_ID , SALARY , AVERAGE SALARY PAID TO THE CORRESPONDING MANAGER_ID

SELECT FIRST_NAME, MANAGER_ID, SALARY, TRUNC("AVG(SALARY)")
FROM EMPLOYEES JOIN (SELECT AVG(SALARY), MANAGER_ID FROM EMPLOYEES GROUP BY MANAGER_ID) USING(MANAGER_ID);

4 DISPLAY FIRST_NAME , HIRE YEAR , SALARY , HIRE YEAR WISE LOWEST SALARY , HIRE YEAR WISE HIGHEST SALARY

SELECT FIRST_NAME , HIRE_YEAR , "MIN(SALARY)" , "MAX(SALARY)" FROM 
	(SELECT FIRST_NAME , TO_CHAR(HIRE_DATE,'RRRR') HIRE_YEAR FROM EMPLOYEES) NATURAL JOIN
	(SELECT MAX(SALARY) , MIN(SALARY) , TO_CHAR(HIRE_DATE,'RRRR') HIRE_YEAR
FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,'RRRR'));

SELECT FIRST_NAME , TO_CHAR(HIRE_DATE,'YYYY') , SALARY , MIN_SALARY , MAX_SALARY 
	FROM EMPLOYEES E
	JOIN (SELECT  TO_CHAR(HIRE_DATE,'YYYY') HIRE_YEAR , MIN(SALARY) MIN_SALARY , MAX(SALARY) MAX_SALARY FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,'YYYY')) O
	ON TO_CHAR(E.HIRE_DATE,'YYYY') = O.HIRE_YEAR ;

5 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPARTMENT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY

SELECT FIRST_NAME, SALARY, DEPARTMENT_ID, "DEPARTMENT WISE HIGHEST SALARY", JOB_ID, "JOB WISE LOWEST SALARY"
FROM EMPLOYEES 
NATURAL JOIN (SELECT DEPARTMENT_ID, MAX(SALARY) "DEPARTMENT WISE HIGHEST SALARY" FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID, MIN(SALARY) "JOB WISE LOWEST SALARY" FROM EMPLOYEES GROUP BY JOB_ID);

6 DISPLAY LAST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , HIGHEST SALARY FOR THE CORRESPONDING DEPARTMENT_ID

SELECT LAST_NAME, DEPARTMENT_ID, SALARY, "LOWEST SALARY", "HIGHEST SALARY"
FROM EMPLOYEES 
NATURAL JOIN (SELECT MIN(SALARY) "LOWEST SALARY", MAX(SALARY) "HIGHEST SALARY", DEPARTMENT_ID FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

7 DISPLAY FIRST_NAME , JOB_ID , SALARY , HIGHEST SALARY , DIFFERENCE (HIGHEST SALARY - SALARY) FOR THE CORRESPONDING JOB_ID

SELECT FIRST_NAME, JOB_ID, SALARY, "HIGHEST SALARY", "HIGHEST SALARY"-SALARY "DIFFERENCE" 
FROM EMPLOYEES 
NATURAL JOIN (SELECT MAX(SALARY) "HIGHEST SALARY" , JOB_ID FROM EMPLOYEES GROUP BY JOB_ID);

8 DISPLAY FIRST_NAME , LAST_NAME , DEPARTMENT_ID , JOB_ID , NO. OF EMPLOYEES WORKING FOR DEPARTMENT_ID , NO. OF EMPLOYEES WORKING FOR JOB_ID

SELECT FIRST_NAME, LAST_NAME, DEPARTMENT_ID, JOB_ID, "NO. OF EMPLOYEES WORKING FOR DEPARTMENT_ID", "NO. OF EMPLOYEES WORKING FOR JOB_ID"
FROM EMPLOYEES 
NATURAL JOIN (SELECT DEPARTMENT_ID, COUNT(*) "NO. OF EMPLOYEES WORKING FOR DEPARTMENT_ID" FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID, COUNT(*) "NO. OF EMPLOYEES WORKING FOR JOB_ID" FROM EMPLOYEES GROUP BY JOB_ID);

9 DISPLAY FIRST_NAME , JOB_ID , SALARY , TOTAL SALARY PAID TO JOB_ID , %OF SALARY EMPLOYEE GET FOR THE CORRESPONDING JOB_ID

SELECT FIRST_NAME, JOB_ID, SALARY, JOB_TOT_SAL , TRUNC(SALARY*100/JOB_TOT_SAL) JOB_PERCENT_SAL FROM EMPLOYEES 
NATURAL JOIN (SELECT JOB_ID,SUM(SALARY) JOB_TOT_SAL FROM EMPLOYEES GROUP BY JOB_ID) ;

10 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , DIFFERENCE IN % FOR THE CORRESPONDING DEPARTMENT_ID

SELECT FIRST_NAME, DEPARTMENT_ID, SALARY, MIN_SAL, TRUNC(SALARY*100/SAL)  DIFF_IN_PER_DEPT
FROM EMPLOYEES 
NATURAL JOIN (SELECT MIN(SALARY) MIN_SAL, SUM(SALARY) SAL, DEPARTMENT_ID  FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

11 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY

SELECT FIRST_NAME, SALARY, DEPARTMENT_ID, DEPT_HIGH_SAL, JOB_ID, JOB_LOW_SAL
FROM EMPLOYEES
NATURAL JOIN (SELECT MAX(SALARY) DEPT_HIGH_SAL , DEPARTMENT_ID FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT MIN(SALARY) JOB_LOW_SAL , JOB_ID FROM EMPLOYEES GROUP BY JOB_ID);


12 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES

SELECT LAST_NAME, SALARY ,DEPARTMENT_ID, DEPAT_AVG_SAL, MANAGER_ID , MGR_NUM_EMP
FROM EMPLOYEES 
NATURAL JOIN (SELECT AVG(SALARY) DEPAT_AVG_SAL, DEPARTMENT_ID FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT COUNT(*) MGR_NUM_EMP,MANAGER_ID FROM EMPLOYEES GROUP BY MANAGER_ID);



13 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES

SELECT LAST_NAME, SALARY, MANAGER_ID, MGR_AVG_SAL, JOB_ID, JOB_NUM_EMP
FROM EMPLOYEES 
NATURAL JOIN (SELECT AVG(SALARY) MGR_AVG_SAL, MANAGER_ID FROM EMPLOYEES GROUP BY MANAGER_ID)
NATURAL JOIN (SELECT COUNT(*) JOB_NUM_EMP, JOB_ID FROM EMPLOYEES GROUP BY JOB_ID);

14 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID 
, JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY

SELECT LAST_NAME, SALARY, DEPARTMENT_ID, DEPT_LOW_SAL, JOB_ID, JOB_AVG_SAL, MANAGER_ID, MGR_HIGH_SAL
FROM EMPLOYEES 
NATURAL JOIN (SELECT MIN(SALARY) DEPT_LOW_SAL, DEPARTMENT_ID FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT AVG(SALARY) JOB_AVG_SAL, JOB_ID FROM EMPLOYEES GROUP BY JOB_ID)
NATURAL JOIN (SELECT MAX(SALARY) MGR_HIGH_SAL, MANAGER_ID FROM EMPLOYEES GROUP BY MANAGER_ID);

15 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , DEPARTMENT_NAME

SELECT LAST_NAME, SALARY, DEPARTMENT_ID, DEPT_HIGH_SAL DEPARTMENT_NAME
FROM EMPLOYEES 
NATURAL JOIN (SELECT MAX(SALARY) DEPT_HIGH_SAL , DEPARTMENT_ID FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

16 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES , 
DEPARTMENT_NAME , CITY

SELECT LAST_NAME, SALARY, DEPARTMENT_ID, DEPT_AVG_SAL, MANAGER_ID, MGR_COUNT_EMP
FROM EMPLOYEES 
NATURAL JOIN (SELECT DEPARTMENT_ID, AVG(SALARY) DEPT_AVG_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT MANAGER_ID, COUNT(*) MGR_COUNT_EMP FROM EMPLOYEES GROUP BY MANAGER_ID);

17 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES , 
DEPARTMENT_NAME , CITY , COUNTRY_NAME

SELECT LAST_NAME, SALARY, MANAGER_ID, MGR_AVG_SAL, JOB_ID, JOB_COUNT_EMP
FROM EMPLOYEES
NATURAL JOIN (SELECT MANAGER_ID, AVG(SALARY) MGR_AVG_SAL FROM EMPLOYEES GROUP BY MANAGER_ID)
NATURAL JOIN (SELECT JOB_ID, COUNT(*) JOB_COUNT_EMP FROM EMPLOYEES GROUP BY JOB_ID);

18 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, 
MANAGER_ID , MANAGER WISE HIGHEST SALARY , DEPARTMENT_NAME , CITY , COUNTRY_NAME , REGION_NAME

SELECT LAST_NAME, SALARY, DEPARTMENT_ID, DEPT_LOW_SAL, JOB_ID, JOB_AVG_SAL, MANAGER_ID, MGR_HIGH_SAL,DEPARTMENT_NAME,CITY, COUNTRY_NAME, REGION_NAME
FROM EMPLOYEES
NATURAL JOIN (SELECT DEPARTMENT_ID, MIN(SALARY) DEPT_LOW_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID, AVG(SALARY) JOB_AVG_SAL FROM EMPLOYEES GROUP BY JOB_ID)
NATURAL JOIN (SELECT MANAGER_ID , MAX(SALARY) MGR_HIGH_SAL FROM EMPLOYEES GROUP BY MANAGER_ID)
NATURAL JOIN DEPARTMENTS
NATURAL JOIN LOCATIONS 
NATURAL JOIN COUNTRIES 
NATURAL JOIN REGIONS;
