1 DISPLAY THE EMPLOYEE INFO. WHO GETS HIGHER SALARY THAN AVERAGE SALARY PAID

	SELECT * FROM EMPLOYEES WHERE SALARY > (SELECT AVG(SALARY) FROM EMPLOYEES);

2 DISPLAY THE EMPLOYEE INFO. WHO WERE HIRED FIRST IN COMPANY

	SELECT * FROM EMPLOYEES WHERE HIRE_DATE = (SELECT MIN(HIRE_DATE) FROM EMPLOYEES);

3 DISPLAY THE EMPLOYEE INFO. WHO WERE HIRED LAST IN COMPANY

	SELECT * FROM EMPLOYEES WHERE HIRE_DATE = (SELECT MAX(HIRE_DATE) FROM EMPLOYEES);

4 DISPLAY INFO. OF EMPLOYEE WHO GETS HIGHEST COMMISSION AMOUNT

	SELECT * FROM EMPLOYEES WHERE COMMISSION_PCT = (SELECT MAX(COMMISSION_PCT) FROM EMPLOYEES);

5 DISPLAY INFO. OF EMPLOYEE WHO GETS LOWEST COMMISSION AMOUNT

	SELECT * FROM EMPLOYEES WHERE COMMISSION_PCT = (SELECT MIN(COMMISSION_PCT) FROM EMPLOYEES);

6 DISPLAY THOSE WHO GETS LOWER SALARY THAN AVERAGE SALARY PAID TO IT_PROG

	SELECT * FROM EMPLOYEES WHERE SALARY < (SELECT AVG(SALARY) FROM EMPLOYEES WHERE JOB_ID='IT_PROG');

7 DISPLAY THOSE WHO GETS HIGHER SALARY THAN HIGHEST SALARY PAID TO DEPARTMENT_ID 80

	SELECT * FROM EMPLOYEES WHERE SALARY > (SELECT MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=80);

8 DISPLAY THOSE WHO GETS LOWER SALARY THAN ALL EMPLOYEES OF DEPARTMENT_ID 60

	SELECT * FROM EMPLOYEES WHERE SALARY < (SELECT MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=60);

9 DISPLAY THOSE WHO GETS HIGHER SALARY THAN ANY SA_REP

	SELECT * FROM EMPLOYEES WHERE SALARY > (SELECT MIN(SALARY) FROM EMPLOYEES WHERE JOB_ID='SA_REP');

10 DISPLAY THOSE WHO GETS LOWER SALARY THAN LOWEST SALARY PAID WHO REPORTS TO 149 (USE MULTI ROW COMPARISON OPERATOR)

	SELECT * FROM EMPLOYEES WHERE SALARY < ALL(SELECT MIN(SALARY) FROM EMPLOYEES WHERE MANAGER_ID=149);

11 DISPLAY THOSE WHO GETS HIGHER SALARY THAN HIGHEST SALARY PAID TO DEPARTMENT_ID 20 (USE MULTI ROW COMPARISON OPERATOR)

	SELECT * FROM EMPLOYEES WHERE SALARY > ANY(SELECT MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=20)

12 DISPLAY ALL OTHER THAN THOSE WHO GETS HIGHEST SALARY IN THEIR DEPARTMENT

	SELECT * FROM EMPLOYEES WHERE (SALARY,DEPARTMENT_ID) NOT IN (SELECT MAX(SALARY),DEPARTMENT_ID FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

13 DISPLAY ALL OTHER THAN THOSE WHO GETS LOWEST SALARY FOR THEIR JOB

	SELECT * FROM EMPLOYEES WHERE (SALARY,JOB_ID) NOT IN (SELECT MIN(SALARY),JOB_ID FROM EMPLOYEES GROUP BY JOB_ID);

14 DISPLAY ALL OTHER THAN THOSE WHO GETS EITHER HIGHEST OR LOWER SALARY FOR THEIR MANAGER

	SELECT * FROM EMPLOYEES WHERE 
        (SALARY,MANAGER_ID) NOT IN (SELECT MIN(SALARY),MANAGER_ID FROM EMPLOYEES GROUP BY MANAGER_ID)
		OR
        (SALARY,MANAGER_ID) NOT IN (SELECT MAX(SALARY),MANAGER_ID FROM EMPLOYEES GROUP BY MANAGER_ID);
