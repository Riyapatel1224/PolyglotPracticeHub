1	COUNT TOTAL NO. OF EMPLOYEES WORK IN COMPANY

	SELECT COUNT(*) FROM EMPLOYEES;

2	DISPLAY HIGHEST SALARY PAID FIGURE

	SELECT MAX(SALARY) FROM EMPLOYEES;

3	DISPLAY THE DATE ON WHICH FIRST EMPLOYEE WAS HIRED

	SELECT MIN(HIRE_DATE) FROM EMPLOYEES;

4   DISPLAY FOR HOW MANY UNIQUE JOBS COMPANY HAS HIRED EMPLOYEES.

	SELECT COUNT(DISTINCT JOB_ID) FROM EMPLOYEES;

5   HOW MANY MANAGERS ARE WORKING IN COMPANY.

	SELECT COUNT(DISTINCT MANAGER_ID) FROM EMPLOYEES;

6   FIND AVERAGE SALARY PAID BY COMPANY

	SELECT AVG(SALARY) FROM EMPLOYEES;

7   FIND THE DIFFERENCE BETWEEN HIGHEST AND LOWEST SALARY PAID BY COMPANY.

	SELECT MAX(SALARY)-MIN(SALARY) FROM EMPLOYEES;

8   FIND THE HIGHEST COMMISSION_PCT

	SELECT MAX(COMMISSION_PCT) FROM EMPLOYEES;

9   FIND THE LOWEST COMMISSION AMOUNT PAID BY COMPANY

	SELECT MIN(COMMISSION_PCT) FROM EMPLOYEES;

10	DISPLAY AVERAGE SALARY PAID TO DEPARTMENT 60

	SELECT AVG(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=60;

11	DISPLAY LOWEST SALARY PAID TO IT_PROG

	SELECT MIN(SALARY) FROM EMPLOYEES WHERE JOB_ID='IT_PROG';

12	DISPLAY COUNT OF EMPLOYEES WHO REPORT TO MANAGER_ID 100

	SELECT COUNT(EMPLOYEE_ID) FROM EMPLOYEES WHERE MANAGER_ID=100;

13	DISPLAY COUNT OF EMPLOYEES WHO WERE HIRED BETWEEN YEAR 91 AND 95

	SELECT COUNT(EMPLOYEE_ID) FROM EMPLOYEES WHERE TO_CHAR(HIRE_DATE,'YY') BETWEEN 91 AND 95;

14	DISPLAY COUNT OF EMPLOYEES WHO GETS SALARY UPTO 6000

	SELECT COUNT(EMPLOYEE_ID) FROM EMPLOYEES WHERE SALARY <= 6000;

15	COUNT DISTINCT JOB_ID

	SELECT COUNT(DISTINCT JOB_ID) FROM EMPLOYEES;

16	COUNT UNIQUE MANAGER_ID

	SELECT COUNT(UNIQUE MANAGER_ID) FROM EMPLOYEES;

17	DISPLAY LOWEST AND HIGHEST SALARY PAID TO DEPARTMENT_ID 50

	SELECT MIN(SALARY),MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID = 50;

18	DISPLAY TOTAL SALARY AMOUNT PAID TO ALL ST_CLERK

	SELECT SUM(SALARY) FROM EMPLOYEES WHERE JOB_ID = 'ST_CLERK';

19	COUNT EMPLOYEES WHOSE FIRST_NAME CONTAINS e

	SELECT COUNT(FIRST_NAME) FROM EMPLOYEES WHERE FIRST_NAME LIKE '%e%';

20	COUNT EMPLOYEES WHOSE LAST_NAME EITHER STARTS OR ENDS WITH Z OR z

	SELECT COUNT(LAST_NAME) FROM EMPLOYEES WHERE LAST_NAME LIKE 'Z%' OR LAST_NAME LIKE '%z';

21	COUNT EMPLOYEES WHO WERE HIRED IN EITHER MAY OR MAR OF ANY YEAR
	
	SELECT COUNT(HIRE_DATE) FROM EMPLOYEES WHERE TO_CHAR(HIRE_DATE,'MON')= 'MAY' OR TO_CHAR(HIRE_DATE,'MON')='MAR';

22	COUNT EMPLOYEES WHOSE JOB_ID STARTS WITH ST

	SELECT COUNT(EMPLOYEE_ID) FROM EMPLOYEES WHERE JOB_ID LIKE 'ST%';

23	COUNT EMPLOYEES WHOSE JOB_ID ENDS WITH MAN

	SELECT COUNT(EMPLOYEE_ID) FROM EMPLOYEES WHERE JOB_ID LIKE '%MAN';

24	COUNT EMPLOYEES WHO DOESN'T GET ANY COMMISSION

	SELECT COUNT(*)-COUNT(COMMISSION_PCT) FROM EMPLOYEES;

25	COUNT EMPLOYEES WHOSE FIRST_NAME CONTAINS TWO CONSECUTIVE SMALL CASE 'L'

	SELECT COUNT(FIRST_NAME) FROM EMPLOYEES WHERE FIRST_NAME LIKE '%ll%';

26	COUNT EMPLOYEES WHOSE SECOND DIGIT OF EMPLOYEE_ID IS 0 (ZERO)

	SELECT COUNT(EMPLOYEE_ID) FROM EMPLOYEES WHERE EMPLOYEE_ID LIKE '_0%';

27   DISPLAY HIRE_DATE ON WHICH FIRST AND LAST EMPLOYEES WERE HIRED.

	SELECT MIN(HIRE_DATE),MAX(HIRE_DATE) FROM EMPLOYEES;

28   DISPLAY DAYS DIFFERENCE BETWEEN LAST AND FIRST EMPLOYEE HIRED.

	SELECT MAX(HIRE_DATE)-MIN(HIRE_DATE) FROM EMPLOYEES;

28   DISPLAY WEEKS DIFFERENCE BETWEEN LAST AND FIRST EMPLOYEE HIRED.

	SELECT ((MAX(HIRE_DATE)-MIN(HIRE_DATE))/7) FROM EMPLOYEES;

28   DISPLAY MONTHS DIFFERENCE , QUARTER DIFFERENCE , YEAR DIFFERENCE BETWEEN LAST AND FIRST EMPLOYEE HIRED.

	SELECT MAX(TO_CHAR(HIRE_DATE,'MM'))-MIN(TO_CHAR(HIRE_DATE,'MM')), MAX(TO_CHAR(HIRE_DATE,'Q'))-							MIN(TO_CHAR(HIRE_DATE,'Q')),MAX(TO_CHAR(HIRE_DATE,'YYYY'))-MIN(TO_CHAR(HIRE_DATE,'YYYY')) FROM EMPLOYEES;
